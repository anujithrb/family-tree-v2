# Admin App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the admin back-office Angular app (`apps/web-admin/`) with password+TOTP authentication, dashboard, community/user/link CRUD management, tree view with edit capabilities, and audit log viewer.

**Architecture:** The web-admin app consumes the Admin API server (port 3001). It reuses shared-ui components (tree viewer, detail panel, wizard) from `packages/shared-ui/`. Admin auth uses email + password + optional TOTP, different from the user app's magic link flow. The admin app is desktop-focused with basic responsiveness.

**Tech Stack:** Angular v21 (zoneless, signals, standalone), Vitest

**Depends on:** Plan 1 (Foundation) and Plan 2 (User App / shared-ui components) must be completed first.

```typescript
import { Injectable, inject, signal, computed } from '@angular/core';
import { HttpClient, HttpErrorResponse } from '@angular/common/http';
import { Observable, tap, throwError, catchError } from 'rxjs';

interface AdminProfile {
  id: string;
  email: string;
}

interface LoginResponse {
  accessToken: string;
  refreshToken: string;
}

interface TwoFAError {
  requires2FA: boolean;
  error: string;
}

@Injectable({ providedIn: 'root' })
export class AdminAuthService {
  private readonly http = inject(HttpClient);

  private accessToken = signal<string | null>(null);
  private refreshTokenValue = signal<string | null>(null);

  readonly admin = signal<AdminProfile | null>(null);
  readonly isAuthenticated = computed(() => this.accessToken() !== null);

  getAccessToken(): string | null {
    return this.accessToken();
  }

  login(email: string, password: string, otpCode?: string): Observable<LoginResponse> {
    const body: Record<string, string> = { email, password };
    if (otpCode) body['otpCode'] = otpCode;

    return this.http.post<LoginResponse>('/api/auth/login', body).pipe(
      tap((res) => {
        this.accessToken.set(res.accessToken);
        this.refreshTokenValue.set(res.refreshToken);
      }),
      catchError((err: HttpErrorResponse) => {
        if (err.status === 403 && err.error?.requires2FA) {
          return throwError(() => ({ requires2FA: true, error: err.error.error }));
        }
        return throwError(() => err);
      }),
    );
  }

  refreshToken(): Observable<{ accessToken: string }> {
    return this.http
      .post<{ accessToken: string }>('/api/auth/refresh', {
        refreshToken: this.refreshTokenValue(),
      })
      .pipe(
        tap((res) => {
          this.accessToken.set(res.accessToken);
        }),
      );
  }

  loadProfile(): Observable<AdminProfile> {
    return this.http.get<AdminProfile>('/api/auth/me').pipe(
      tap((profile) => this.admin.set(profile)),
    );
  }

  logout(): void {
    this.accessToken.set(null);
    this.refreshTokenValue.set(null);
    this.admin.set(null);
  }
}
```

```typescript
import { inject } from '@angular/core';
import { HttpInterceptorFn } from '@angular/common/http';
import { AdminAuthService } from './admin-auth.service';

export const adminAuthInterceptor: HttpInterceptorFn = (req, next) => {
  const authService = inject(AdminAuthService);

  if (req.url.includes('/api/auth/login') || req.url.includes('/api/auth/refresh')) {
    return next(req);
  }

  const token = authService.getAccessToken();
  if (token) {
    return next(req.clone({ setHeaders: { Authorization: `Bearer ${token}` } }));
  }

  return next(req);
};
```

import { Injectable, inject } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface AdminUser {
  id: string;
  email: string;
  displayName: string;
  status: string;
  createdAt: string;
}

export interface AdminUserFilters {
  status?: string;
  name?: string;
  email?: string;
  page?: number;
}

@Injectable({ providedIn: 'root' })
export class AdminUserApiService {
  private readonly http = inject(HttpClient);

  list(filters?: AdminUserFilters): Observable<AdminUser[]> {
    let params = new HttpParams();
    if (filters) {
      Object.entries(filters).forEach(([key, value]) => {
        if (value != null) params = params.set(key, String(value));
      });
    }
    return this.http.get<AdminUser[]>('/api/users', { params });
  }

  getById(id: string): Observable<AdminUser> {
    return this.http.get<AdminUser>(`/api/users/${id}`);
  }

  invite(email: string, displayName: string): Observable<unknown> {
    return this.http.post('/api/users/invite', { email, displayName });
  }

  update(id: string, data: { displayName?: string; status?: string }): Observable<unknown> {
    return this.http.put(`/api/users/${id}`, data);
  }

  remove(id: string): Observable<unknown> {
    return this.http.delete(`/api/users/${id}`);
  }
}

@Component({
  selector: 'fta-login',
  standalone: true,
  imports: [FormsModule],
  templateUrl: './admin-login.component.html',
  styleUrl: './admin-login.component.scss',
})
export class AdminLoginComponent {
  private readonly auth = inject(AdminAuthService);
  private readonly router = inject(Router);

  readonly email = signal('');
  readonly password = signal('');
  readonly otpCode = signal('');
  readonly requires2FA = signal(false);
  readonly loading = signal(false);
  readonly error = signal<string | null>(null);

  onSubmit(): void {
    this.loading.set(true);
    this.error.set(null);

    const otp = this.requires2FA() ? this.otpCode() : undefined;
    this.auth.login(this.email(), this.password(), otp).subscribe({
      next: () => {
        this.loading.set(false);
        this.router.navigate(['/dashboard']);
      },
      error: (err) => {
        this.loading.set(false);
        if (err?.requires2FA) {
          this.requires2FA.set(true);
        } else {
          this.error.set(err?.error?.message ?? 'Login failed');
        }
      },
    });
  }
}

import { Component, inject, OnInit, signal } from '@angular/core';
import { RouterLink } from '@angular/router';
import { FormsModule } from '@angular/forms';
import { AdminUserApiService, AdminUser } from '../../../core/api/admin-user-api.service';

@Component({
  selector: 'fta-user-list',
  standalone: true,
  imports: [RouterLink, FormsModule],
  templateUrl: './user-list.component.html',
  styleUrl: './user-list.component.scss',
})
export class UserListComponent implements OnInit {
  private readonly userApi = inject(AdminUserApiService);

  readonly users = signal<AdminUser[]>([]);
  readonly loading = signal(false);
  readonly filterStatus = signal('');
  readonly filterName = signal('');

  ngOnInit(): void {
    this.loadUsers();
  }

  loadUsers(): void {
    this.loading.set(true);
    const filters: Record<string, string> = {};
    if (this.filterStatus()) filters['status'] = this.filterStatus();
    if (this.filterName()) filters['name'] = this.filterName();
    this.userApi.list(filters).subscribe({
      next: (list) => { this.users.set(list); this.loading.set(false); },
      error: () => this.loading.set(false),
    });
  }

  onDelete(id: string): void {
    this.userApi.remove(id).subscribe(() => this.loadUsers());
  }

  onInvite(email: string, displayName: string): void {
    this.userApi.invite(email, displayName).subscribe(() => this.loadUsers());
  }
}

export class LinkListComponent implements OnInit {

  }
