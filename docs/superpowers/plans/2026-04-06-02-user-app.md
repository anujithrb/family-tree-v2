# User App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the user-facing Angular app (`apps/web-user/`) with authentication, tree visualization, detail panel, community creation wizard, relationship finder, and user profile — all powered by shared components in `packages/shared-ui/`.

**Architecture:** The web-user app consumes the API server (port 3000) via HTTP client services. State is managed with Angular signals — no store library. Shared-ui provides the tree viewer, detail panel, and wizard components. The app uses magic-link auth with JWT access + refresh tokens.

**Tech Stack:** Angular v21 (zoneless, signals, standalone), D3 (zoom/pan only), Vitest

**Depends on:** Plan 1 (Frontend Foundation) must be completed first.

```typescript
import { Injectable, inject, signal, computed } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable, tap } from 'rxjs';
import type { UserProfile } from '@family-tree/shared-ui';

interface InviteResponse {
  message: string;
  magicLinkToken: string;
  userId: string;
}

interface VerifyResponse {
  accessToken: string;
  refreshToken: string;
}

interface RefreshResponse {
  accessToken: string;
}

@Injectable({ providedIn: 'root' })
export class AuthService {
  private readonly http = inject(HttpClient);

  private accessToken = signal<string | null>(null);
  private refreshTokenValue = signal<string | null>(null);

  readonly user = signal<UserProfile | null>(null);
  readonly isAuthenticated = computed(() => this.accessToken() !== null);

  getAccessToken(): string | null {
    return this.accessToken();
  }

  sendInvite(email: string, displayName: string): Observable<InviteResponse> {
    return this.http.post<InviteResponse>('/api/auth/invite', { email, displayName });
  }

  verifyMagicLink(token: string): Observable<VerifyResponse> {
    return this.http.get<VerifyResponse>(`/api/auth/verify/${token}`).pipe(
      tap((res) => {
        this.accessToken.set(res.accessToken);
        this.refreshTokenValue.set(res.refreshToken);
      }),
    );
  }

  refreshToken(): Observable<RefreshResponse> {
    return this.http
      .post<RefreshResponse>('/api/auth/refresh', {
        refreshToken: this.refreshTokenValue(),
      })
      .pipe(
        tap((res) => {
          this.accessToken.set(res.accessToken);
        }),
      );
  }

  loadProfile(): Observable<UserProfile> {
    return this.http.get<UserProfile>('/api/users/me').pipe(
      tap((profile) => {
        this.user.set(profile);
      }),
    );
  }

  logout(): void {
    this.accessToken.set(null);
    this.refreshTokenValue.set(null);
    this.user.set(null);
  }
}
```

```typescript
import { inject } from '@angular/core';
import { HttpInterceptorFn } from '@angular/common/http';
import { AuthService } from './auth.service';

export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const authService = inject(AuthService);

  // Skip auth header for auth endpoints
  if (req.url.includes('/api/auth/')) {
    return next(req);
  }

  const token = authService.getAccessToken();
  if (token) {
    const cloned = req.clone({
      setHeaders: { Authorization: `Bearer ${token}` },
    });
    return next(cloned);
  }

  return next(req);
};
```

```typescript
import { inject } from '@angular/core';
import { CanActivateFn, Router } from '@angular/router';

export const authGuard: CanActivateFn = () => {
  const authService = inject(AuthService);

  if (authService.isAuthenticated()) {
    return true;
```

```typescript
import { Injectable, inject, signal, computed } from '@angular/core';

@Injectable({ providedIn: 'root' })
export class CommunityState {
  private readonly communityApi = inject(CommunityApiService);

  readonly communities = signal<Community[]>([]);
  readonly activeId = signal<string | null>(null);
  readonly loading = signal(false);

  readonly active = computed(() => {
    const id = this.activeId();
    return this.communities().find((c) => c.id === id) ?? null;
  });

  loadCommunities(): void {
    this.loading.set(true);
    this.communityApi.list().subscribe({
      next: (list) => {
        this.communities.set(list);
        if (list.length > 0 && !this.activeId()) {
          this.activeId.set(list[0].id);
        }
        this.loading.set(false);
      },
      error: () => {
        this.loading.set(false);
      },
    });
  }

  setActive(communityId: string): void {
    this.activeId.set(communityId);
  }
}
```

```typescript
import { Injectable, inject, signal } from '@angular/core';
import { TreeApiService } from '../api/tree-api.service';
import type { TreeResponse, TreePerson } from '@family-tree/shared-ui';

@Injectable({ providedIn: 'root' })
export class TreeState {
  private readonly treeApi = inject(TreeApiService);

  readonly treeData = signal<TreeResponse | null>(null);
  readonly selectedNode = signal<TreePerson | null>(null);
  readonly loading = signal(false);
  readonly error = signal<string | null>(null);

  loadTree(communityId: string): void {
    this.loading.set(true);
    this.error.set(null);
    this.treeApi.getTree(communityId).subscribe({
      next: (data) => {
        this.treeData.set(data);
        this.loading.set(false);
        // Preserve selection if same person still exists
        const current = this.selectedNode();
        if (current) {
          const stillExists = data.people.find((p) => p.nodeId === current.nodeId);
          if (stillExists) {
            this.selectedNode.set(stillExists);
          } else {
            this.selectedNode.set(null);
          }
        }
      },
      error: (err) => {
        this.error.set(err?.error?.error ?? 'Failed to load tree');
        this.loading.set(false);
      },
    });
  }

  selectNode(person: TreePerson): void {
    this.selectedNode.set(person);
  }

  clearSelection(): void {
    this.selectedNode.set(null);
  }
}
```

```typescript
@Component({
  selector: 'g[ft-tree-canvas]',

  templateUrl: './tree-canvas.component.html',

export class TreeCanvasComponent {
  readonly treeData = input.required<TreeResponse>();
  readonly selectedNodeId = input<string | null>(null);

  readonly nodeSelected = output<TreePerson>();

  private readonly layoutService = new TreeLayoutService();
  private readonly C = DEFAULT_LAYOUT_CONSTANTS;

  readonly layout = computed(() => {
    const data = this.treeData();
    if (!data || data.couples.length === 0) {
      return null;
    }

    const layoutCouples: LayoutCouple[] = data.couples.map((c) => ({
      id: c.id,
      spouseAId: c.spouseAId,
      spouseBId: c.spouseBId,
      children: c.children,
    }));

    const personCoupleMap: Record<string, string> = {};
    for (const couple of data.couples) {
      for (const childId of couple.children) {
        const childHasCouple = data.couples.find(
          (c) => c.spouseAId === childId || c.spouseBId === childId,
        );
        if (childHasCouple) {
          personCoupleMap[childId] = childHasCouple.id;
        }
      }
    }

    return this.layoutService.computeLayout(layoutCouples, personCoupleMap);
  });

  readonly personMap = computed(() => {
    const map = new Map<string, TreePerson>();
    for (const p of this.treeData().people) {
      map.set(p.nodeId, p);
    }
    return map;
  });

  readonly coupleEntries = computed(() => {
    const l = this.layout();
    if (!l) return [];
    return Object.entries(l.couples);
  });

  readonly soloEntries = computed(() => {
    const l = this.layout();
    if (!l) return [];
    return Object.entries(l.soloNodes);
  });

  getPerson(nodeId: string): TreePerson | undefined {
    return this.personMap().get(nodeId);
  }

  getCouple(coupleId: string): { spouseAId: string; spouseBId: string; children: string[] } | undefined {
    return this.treeData().couples.find((c) => c.id === coupleId);
  }

  onPersonSelected(person: TreePerson): void {
    this.nodeSelected.emit(person);
  }
}
```

```typescript
import { Injectable, signal } from '@angular/core';
import { zoom, zoomIdentity, ZoomBehavior, ZoomTransform } from 'd3-zoom';
import { select, Selection } from 'd3-selection';

@Injectable({ providedIn: 'root' })
export class TreeZoomService {
  readonly transform = signal('translate(0,0) scale(1)');

  private zoomBehavior: ZoomBehavior<SVGSVGElement, unknown> | null = null;
  private svgSelection: Selection<SVGSVGElement, unknown, null, undefined> | null = null;
  private currentTransform: ZoomTransform = zoomIdentity;

  attach(svgElement: SVGSVGElement): void {
    this.svgSelection = select(svgElement);
    this.zoomBehavior = zoom<SVGSVGElement, unknown>()
      .scaleExtent([0.1, 4])
      .on('zoom', (event) => {
        this.currentTransform = event.transform;
        this.transform.set(
          `translate(${event.transform.x},${event.transform.y}) scale(${event.transform.k})`,
        );
      });

    this.svgSelection.call(this.zoomBehavior);
  }

  detach(): void {
    if (this.svgSelection) {
      this.svgSelection.on('.zoom', null);
    }
    this.svgSelection = null;
    this.zoomBehavior = null;
  }

  zoomIn(): void {
    if (this.svgSelection && this.zoomBehavior) {
      this.svgSelection.call(this.zoomBehavior.scaleBy, 1.3);
    }
  }

  zoomOut(): void {
    if (this.svgSelection && this.zoomBehavior) {
      this.svgSelection.call(this.zoomBehavior.scaleBy, 0.7);
    }
  }

  fitToScreen(canvasWidth: number, canvasHeight: number): void {
    if (!this.svgSelection || !this.zoomBehavior) return;

    const svgEl = this.svgSelection.node();
    if (!svgEl) return;

    const { width, height } = svgEl.getBoundingClientRect();
    if (width === 0 || height === 0) return;

    const scale = Math.min(width / canvasWidth, height / canvasHeight) * 0.9;
    const tx = (width - canvasWidth * scale) / 2;
    const ty = (height - canvasHeight * scale) / 2;

    this.svgSelection.call(
      this.zoomBehavior.transform,
      zoomIdentity.translate(tx, ty).scale(scale),
    );
  }

  panTo(x: number, y: number): void {
    if (!this.svgSelection || !this.zoomBehavior) return;

    const svgEl = this.svgSelection.node();
    if (!svgEl) return;

    const { width, height } = svgEl.getBoundingClientRect();
    const k = this.currentTransform.k;

    this.svgSelection.call(
      this.zoomBehavior.transform,
      zoomIdentity.translate(width / 2 - x * k, height / 2 - y * k).scale(k),
    );
  }
}
```

```typescript
import { Injectable, signal, computed } from '@angular/core';
import type { PanelState, PanelMode } from '../models/panel.model';
import type { TreePerson } from '../../tree-viewer/models/tree-data.model';

@Injectable({ providedIn: 'root' })
export class PanelNavigationService {
  private readonly history = signal<PanelState[]>([]);

  readonly current = computed<PanelState | null>(() => {
    const h = this.history();
    return h.length > 0 ? h[h.length - 1] : null;
  });

  readonly isOpen = computed(() => this.history().length > 0);

  readonly canGoBack = computed(() => this.history().length > 1);

  openPerson(person: TreePerson): void {
    this.history.set([{ person, mode: 'view' }]);
  }

  navigateToPerson(person: TreePerson): void {
    this.history.update((h) => [...h, { person, mode: 'view' }]);
  }

  pushMode(mode: PanelMode): void {
    const curr = this.current();
    if (!curr) return;
    this.history.update((h) => [...h, { person: curr.person, mode }]);
  }

  goBack(): void {
    this.history.update((h) => {
      if (h.length <= 1) return h;
      return h.slice(0, -1);
    });
  }

  close(): void {
    this.history.set([]);
  }
}
```

```typescript
    this.formGender.set(p.gender ?? '');
    this.formBirthYear.set(p.birthYear);
    this.formDeathYear.set(p.deathYear);

    this.initForm();
    this.modeChange.emit('edit');

    if (mode === 'edit') this.initForm();
    this.modeChange.emit(mode);
  }

  onSubmit(): void {
    const mode = this.mode();

    if (mode === 'edit') {
      this.formSubmit.emit({
        action: 'edit',
        nodeId: this.person().nodeId,
        name: this.formName(),
        gender: this.formGender() || undefined,
        birthYear: this.formBirthYear(),
        deathYear: this.formDeathYear(),
        isDeceased: this.formIsDeceased(),
      });
    } else if (mode === 'add-child') {
      this.formSubmit.emit({
        action: 'add-child',
        coupleId: this.personCouple()?.id,
        name: this.formName(),
        gender: this.formGender() || undefined,
        birthYear: this.formBirthYear(),
      });
    } else if (mode === 'add-spouse') {
      this.formSubmit.emit({
        action: 'add-spouse',
        treeNodeId: this.person().nodeId,
        name: this.formName(),
        gender: this.formGender() || undefined,
        birthYear: this.formBirthYear(),
      });
    } else if (mode === 'add-sibling') {
      this.formSubmit.emit({
        action: 'add-sibling',
        treeNodeId: this.person().nodeId,
        name: this.formName(),
        gender: this.formGender() || undefined,
        birthYear: this.formBirthYear(),
      });
    } else if (mode === 'add-parents') {
      this.formSubmit.emit({
        action: 'add-parents',
        treeNodeId: this.person().nodeId,
        parent1: { name: this.parent1Name(), gender: this.parent1Gender() || undefined, birthYear: this.parent1BirthYear() },
        parent2: { name: this.parent2Name(), gender: this.parent2Gender() || undefined, birthYear: this.parent2BirthYear() },
      });
    }
  }

    this.personNavigate.emit(nodeId);
```

```typescript
import { signal, computed } from '@angular/core';
import type {
  TempNode,
  TempCouple,
  TempChild,
  WizardConfig,
  WizardSubmission,
} from '../models/wizard.model';

export class WizardStateService {
  readonly currentStep = signal(1);
  readonly communityName = signal('');
  readonly nodes = signal<TempNode[]>([]);
  readonly couples = signal<TempCouple[]>([]);
  readonly children = signal<TempChild[]>([]);
  readonly selectedTempId = signal<string | null>(null);
  readonly adminNodeTempId = signal<string | null>(null);

  readonly isValid = computed(() => {
    return this.communityName().trim().length > 0 && this.nodes().length >= 2;
  });

  private nextTempId = 1;

  generateTempId(): string {
    return `temp-${this.nextTempId++}`;
  }

  initWithConfig(config: WizardConfig): void {
    if (config.selfNode) {
      this.nodes.set([config.selfNode]);
    }
  }

  setCommunityName(name: string): void {
    this.communityName.set(name);
  }

  addNode(node: TempNode): void {
    this.nodes.update((n) => [...n, node]);
  }

  removeNode(tempId: string): void {
    this.nodes.update((n) => n.filter((x) => x.tempId !== tempId));
    this.couples.update((c) =>
      c.filter((x) => x.spouse1 !== tempId && x.spouse2 !== tempId),
    );
    this.children.update((ch) =>
      ch.filter(
        (x) =>
          x.childRef !== tempId &&
          x.coupleSpouse1 !== tempId &&
          x.coupleSpouse2 !== tempId,
      ),
    );
  }

  addCouple(spouse1TempId: string, spouse2TempId: string): void {
    this.couples.update((c) => [
      ...c,
      { spouse1: spouse1TempId, spouse2: spouse2TempId },
    ]);
  }

  addChildToCouple(
    coupleSpouse1: string,
    coupleSpouse2: string,
    childRef: string,
  ): void {
    const sortOrder = this.children().filter(
      (ch) =>
        ch.coupleSpouse1 === coupleSpouse1 &&
        ch.coupleSpouse2 === coupleSpouse2,
    ).length;
    this.children.update((ch) => [
      ...ch,
      { coupleSpouse1, coupleSpouse2, childRef, sortOrder },
    ]);
  }

  nextStep(): void {
    this.currentStep.update((s) => Math.min(s + 1, 3));
  }

  prevStep(): void {
    this.currentStep.update((s) => Math.max(s - 1, 1));
  }

  buildSubmission(): WizardSubmission {
    return {
      name: this.communityName(),
      nodes: this.nodes(),
      couples: this.couples(),
      children: this.children(),
      adminNodeTempId: this.adminNodeTempId() ?? undefined,
    };
  }
}
```

```typescript
import { Component, inject, signal } from '@angular/core';

@Component({
  selector: 'ft-community-switcher',

  templateUrl: './community-switcher.component.html',
  styleUrl: './community-switcher.component.scss',
})
export class CommunitySwitcherComponent {
  readonly communityState = inject(CommunityState);
  private readonly router = inject(Router);

    this.isOpen.update((v) => !v);
  }

  select(communityId: string): void {
    this.communityState.setActive(communityId);
    this.isOpen.set(false);
    this.router.navigate(['/communities', communityId, 'tree']);
  }
}
```

```typescript
  standalone: true,
  imports: [FormsModule],
  templateUrl: './login.component.html',
  styleUrl: './login.component.scss',
})
export class LoginComponent {
  private readonly auth = inject(AuthService);

  readonly email = signal('');
  readonly displayName = signal('');
  readonly sent = signal(false);
  readonly loading = signal(false);
  readonly error = signal<string | null>(null);

  onSubmit(): void {
    this.loading.set(true);
```

```typescript
import { Component, inject, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { AuthService } from '../../../core/auth/auth.service';
import { CommunityState } from '../../../core/state/community.state';

@Component({
  selector: 'ft-verify',
  standalone: true,
  template: `
    <div class="verify-page">
      @if (error()) {
        <h1>Verification Failed</h1>
        <p>{{ error() }}</p>
      } @else {
        <p>Verifying your magic link...</p>
      }
    </div>
  `,
  styles: `
    .verify-page { display: flex; flex-direction: column; align-items: center; justify-content: center; min-height: 100%; }
  `,
})
export class VerifyComponent implements OnInit {
  private readonly route = inject(ActivatedRoute);
  private readonly router = inject(Router);
  private readonly auth = inject(AuthService);
  private readonly communityState = inject(CommunityState);

  readonly error = this.auth.user; // reuse signal slot, replace with dedicated if needed

  ngOnInit(): void {
    const token = this.route.snapshot.paramMap.get('token');
    if (!token) {
      this.router.navigate(['/auth/login']);
      return;
    }

    this.auth.verifyMagicLink(token).subscribe({
      next: () => {
        this.auth.loadProfile().subscribe(() => {
          this.communityState.loadCommunities();
          // Simple redirect — in production, wait for communities to load
          setTimeout(() => {
            const communities = this.communityState.communities();
            if (communities.length === 0) {
              this.router.navigate(['/wizard']);
            } else {
              this.router.navigate(['/communities', communities[0].id, 'tree']);
            }
          }, 500);
        });
      },
      error: () => {
        this.router.navigate(['/auth/login']);
      },
    });
  }
}
```
